# Create a function to plot scatter plot for continent and find and plot regression line
def plot(x_axis,y_axis,name):
    # find the regression line for the group   
    m, b = np.polyfit(x_axis, y_axis, 1)
    
    # construct the plot
    fig, ax = plt.subplots(figsize=(10,5))
    # plot the points for the scatter plot
    ax.scatter(x_axis,y_axis)
    # plot the regression lin
    ax.plot(x_axis, (m*np.array(x_axis)+b), "-", color="red", label='_nolegend_')
    # set labels and title
    ax.set_xlabel("Year")
    ax.set_ylabel("Life Expectancy")
    ax.set_title("Life Expectancy vs. Time grouped by Year for "+name)
    plt.show()

# apply function to each continent in group
df.groupby("continent").apply(lambda x: plot(x["year"],x["lifeExp"],x.name))
